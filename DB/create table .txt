-- 회원 목록
-- DROP TABLE MEMBERS_TBL;
CREATE SEQUENCE SEQ_MEMBERS;

CREATE TABLE MEMBERS_TBL
(
      IDX        NUMBER(5)    NOT NULL PRIMARY KEY
    , MID        VARCHAR2(16) NOT NULL PRIMARY KEY
    , MEM_PWD    VARCHAR2(10) NOT NULL
    , MEM_NAME   VARCHAR2(30) NOT NULL
    , MEM_BIRTH  DATE         NOT NULL
    , MEM_GEN    VARCHAR2(10) NOT NULL
    , MEM_MAIL   VARCHAR2(30) NOT NULL
    , LVL        NUMBER(5)    NOT NULL
    , MEM_DELNUM NUMBER(5)    NOT NULL
);


-- 아티스트 목록

(

    , ART_CNT   NUMBER(10)    NOT NULL
